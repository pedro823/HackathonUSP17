create table post(
id int not null primary key auto_increment,
categoria text,
titulo text,
pesquisador text,
pesquisadorId int,
vagas int,
instituicao text,
status text,
dataInicio date,
dataFim date,
fomento text,
requisitos text,
tags text,
cidade text,
area text,
resumo text
);

create table users(
id int not null primary key auto_increment,
nome text,
email text,
senha text,
tipo text,
universidade text
);
